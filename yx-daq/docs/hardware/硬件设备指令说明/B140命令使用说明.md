# B140 运动控制器命令控制接口说明

## 概述

本文档描述 B140 运动控制器（Galil DMC-B140-M）的命令控制接口，包括底层TCP协议命令格式、代码封装实现、以及如何在项目中扩展新的命令控制功能。

---

## 一、通信基础

### 1.1 连接参数

| 参数 | 默认值 | 说明 |
|------|--------|------|
| 协议 | TCP/IP | Galil DMC 协议 |
| IP地址 | 192.168.1.101 | 设备IP（可配置） |
| 端口 | 5000 | 默认TCP端口 |
| 超时 | 5000ms | 连接/命令超时 |
| 轴数 | 4 | A、B、C、D 四轴 |

### 1.2 数据包格式

**请求命令格式：**
```
<命令字符串>\r
```

**响应格式：**
```
<data>:     成功响应（冒号结尾）
或
<error>?    错误响应（问号结尾）
```

### 1.3 响应类型

| 响应 | 含义 |
|------|------|
| `:` | 命令执行成功（空响应） |
| `<data>:` | 成功并返回数据 |
| `?` | 命令执行失败 |
| `<error>?` | 失败并返回错误信息 |

### 1.4 轴定义

| 轴标识 | 说明 |
|--------|------|
| A | 第1轴 |
| B | 第2轴 |
| C | 第3轴 |
| D | 第4轴 |

**轴掩码（多轴操作）：**
- `AB` - A轴和B轴
- `ABCD` - 所有轴

### 4.1 命令构建器

```typescript
/**
 * B140命令构建器类
 * 用于构建各种运动控制命令
 */
export class B140CommandBuilder {
  /**
   * 绝对位置命令
   * 单轴：PAA=position
   * 多轴：PA position,position,...
   */
  static positionAbsolute(axis: B140Axis | B140AxisMask, position: number): string {
    const roundedPos = Math.round(position)
    return this.isSingleAxis(axis)
      ? `PA${axis}=${roundedPos}`
      : `PA ${roundedPos}`
  }

  /**
   * 相对位置命令
   */
  static positionRelative(axis: B140Axis | B140AxisMask, distance: number): string {
    const roundedDist = Math.round(distance)
    return this.isSingleAxis(axis)
      ? `PR${axis}=${roundedDist}`
      : `PR ${roundedDist}`
  }

  /**
   * 速度命令
   */
  static speed(axis: B140Axis | B140AxisMask, speed: number): string {
    const roundedSpeed = Math.round(speed)
    return this.isSingleAxis(axis)
      ? `SP${axis}=${roundedSpeed}`
      : `SP ${roundedSpeed}`
  }

  /**
   * 加速度命令
   */
  static acceleration(axis: B140Axis | B140AxisMask, accel: number): string {
    const roundedAccel = Math.round(accel)
    return this.isSingleAxis(axis)
      ? `AC${axis}=${roundedAccel}`
      : `AC ${roundedAccel}`
  }

  /**
   * 减速度命令
   */
  static deceleration(axis: B140Axis | B140AxisMask, decel: number): string {
    const roundedDecel = Math.round(decel)
    return this.isSingleAxis(axis)
      ? `DC${axis}=${roundedDecel}`
      : `DC ${roundedDecel}`
  }

  /**
   * 开始运动命令
   */
  static beginMotion(axis?: B140Axis | B140AxisMask): string {
    if (!axis) return 'BG'
    return this.isSingleAxis(axis) ? `BG${axis}` : `BG ${axis}`
  }

  /**
   * 停止运动命令
   */
  static stopMotion(axis?: B140Axis): string {
    return axis ? `ST${axis}` : 'ST'
  }

  /**
   * 紧急停止命令
   */
  static emergencyStop(): string {
    return 'AB'
  }

  /**
   * 回零命令
   */
  static home(axis?: B140Axis): string {
    return axis ? `HM${axis}` : 'HM'
  }

  /**
   * 电机使能命令
   */
  static enable(axis?: B140Axis | B140AxisMask): string {
    if (!axis) return 'SH'
    return this.isSingleAxis(axis) ? `SH${axis}` : `SH ${axis}`
  }

  /**
   * 电机关闭命令
   */
  static disable(axis?: B140Axis | B140AxisMask): string {
    if (!axis) return 'MO'
    return this.isSingleAxis(axis) ? `MO${axis}` : `MO ${axis}`
  }

  /**
   * 查询位置命令
   */
  static tellPosition(axis?: B140Axis): string {
    return axis ? `TD${axis}` : 'TD'
  }

  /**
   * 查询状态命令
   */
  static tellStatus(axis?: B140Axis): string {
    return axis ? `TS${axis}` : 'TS'
  }

  /**
   * 定义位置命令
   */
  static definePosition(axis: B140Axis, position: number): string {
    return `DP${axis}=${Math.round(position)}`
  }

  /**
   * 判断是否为单轴
   */
  private static isSingleAxis(axis: B140Axis | B140AxisMask): boolean {
    return typeof axis === 'string' && axis.length === 1
  }
}
```

### 4.2 响应解析器

```typescript
/**
 * B140响应解析器类
 * 用于解析设备返回的响应数据
 */
export class B140ResponseParser {
  /**
   * 解析多轴位置响应
   * 输入："1000, 2000, 3000, 4000:"
   * 输出：Map { 'A' => 1000, 'B' => 2000, 'C' => 3000, 'D' => 4000 }
   */
  static parseMultiAxisPosition(response: string): Map<string, number> {
    const result = new Map<string, number>()
    const cleanResponse = response.replace(/[:?\r\n]/g, '').trim()
    
    if (!cleanResponse) return result

    const parts = cleanResponse.split(',').map(p => p.trim())
    const axes = ['A', 'B', 'C', 'D']
    
    parts.forEach((part, index) => {
      if (part === '' || index >= axes.length) return
      const value = parseFloat(part)
      if (!isNaN(value)) {
        result.set(axes[index], value)
      }
    })

    return result
  }

  /**
   * 解析单轴位置响应
   * 输入："10000:"
   * 输出：10000
   */
  static parsePosition(response: string): number | null {
    const cleanResponse = response.replace(/[^0-9\-]/g, '').trim()
    if (!cleanResponse) return null
    const value = parseFloat(cleanResponse)
    return isNaN(value) ? null : value
  }

  /**
   * 解析状态响应
   * 输入："128:"
   * 输出：{ moving: true, forwardLimit: false, reverseLimit: false, raw: 128 }
   */
  static parseStatus(response: string): {
    moving: boolean
    forwardLimit: boolean
    reverseLimit: boolean
    raw: number
  } | null {
    const cleanResponse = response.replace(/[^0-9]/g, '').trim()
    if (!cleanResponse) return null

    const value = parseInt(cleanResponse, 10)
    if (isNaN(value)) return null

    return {
      raw: value,
      moving: (value & STATUS_MASK_MOVING) !== 0,
      forwardLimit: (value & STATUS_MASK_FORWARD_LIMIT) !== 0,
      reverseLimit: (value & STATUS_MASK_REVERSE_LIMIT) !== 0
    }
  }

  /**
   * 解析停止代码响应
   * 输入："0:"
   * 输出：0
   */
  static parseStopCode(response: string): number | null {
    const cleanResponse = response.replace(/[^0-9]/g, '').trim()
    if (!cleanResponse) return null
    const value = parseInt(cleanResponse, 10)
    return isNaN(value) ? null : value
  }

  /**
   * 检查响应是否为成功响应
   */
  static isSuccess(response: string): boolean {
    return response.includes(RESPONSE_SUCCESS)
  }

  /**
   * 检查响应是否为错误响应
   */
  static isError(response: string): boolean {
    return response.includes(RESPONSE_ERROR)
  }
}
```

### 4.3 驱动类核心实现

```typescript
/**
 * B140驱动类
 * 封装与B140运动控制器的所有通信
 */
export class B140Driver {
  private connected: boolean = false
  private socket: any = null
  private commandSequence: number = 0
  private pendingCommands: Map<string, PendingCommand> = new Map()
  private responseBuffer: string = ''
  private commandTimeout: number = 5000
  private axisDirections: Map<B140Axis, number> = new Map()

  /**
   * 构造函数
   */
  constructor() {
    // 初始化轴方向（默认正向）
    ;(['A', 'B', 'C', 'D'] as B140Axis[]).forEach(axis => {
      this.axisDirections.set(axis, 1)
    })
  }

  /**
   * 发送命令并等待响应
   * 
   * 流程：
   * 1. 检查连接状态
   * 2. 创建Promise并加入待处理队列
   * 3. 设置超时定时器
   * 4. 发送命令到硬件
   * 5. 等待响应处理
   */
  async sendCommand(command: string): Promise<string> {
    if (!this.connected) {
      throw new Error('设备未连接')
    }

    return new Promise<string>((resolve, reject) => {
      const cmdId = `cmd_${++this.commandSequence}`
      const timer = setTimeout(() => {
        this.pendingCommands.delete(cmdId)
        reject(new Error(`指令超时: ${command}`))
      }, this.commandTimeout)

      this.pendingCommands.set(cmdId, { resolve, reject, timer, command })
      
      // 发送命令（添加结束符）
      this.writeToSocket(command + COMMAND_TERMINATOR)
    })
  }

  /**
   * 向socket写入数据
   */
  private writeToSocket(data: string): void {
    // 实际实现中，这里调用底层TCP socket的write方法
    console.log(`[B140] 发送: ${data.trim()}`)
  }

  /**
   * 处理硬件响应
   * 
   * 响应格式：
   * - 成功：":<data>" 或 "<data>:"
   * - 错误："?<error>" 或 "<error>?"
   */
  private handleResponse(data: string): void {
    this.responseBuffer += data
    
    while (true) {
      const successIndex = this.responseBuffer.indexOf(RESPONSE_SUCCESS)
      const errorIndex = this.responseBuffer.indexOf(RESPONSE_ERROR)

      if (this.pendingCommands.size === 0) {
        if (successIndex !== -1 || errorIndex !== -1) {
          const cutIndex = Math.max(successIndex, errorIndex) + 1
          this.responseBuffer = this.responseBuffer.substring(cutIndex)
          continue
        }
        break
      }

      const firstCmdId = Array.from(this.pendingCommands.keys())[0]
      const pending = this.pendingCommands.get(firstCmdId)!
      
      // 处理错误响应
      if (errorIndex !== -1 && (successIndex === -1 || errorIndex < successIndex)) {
        const errorMsg = this.responseBuffer.substring(0, errorIndex).trim() || 'B140 指令错误'
        pending.reject(new Error(errorMsg))
        this.pendingCommands.delete(firstCmdId)
        clearTimeout(pending.timer)
        this.responseBuffer = this.responseBuffer.substring(errorIndex + 1)
        continue
      }

      // 处理成功响应
      if (successIndex !== -1) {
        const response = this.responseBuffer.substring(0, successIndex).trim()
        pending.resolve(response)
        this.pendingCommands.delete(firstCmdId)
        clearTimeout(pending.timer)
        this.responseBuffer = this.responseBuffer.substring(successIndex + 1)
        continue
      }

      break
    }
  }

  /**
   * 连接设备
   */
  async connect(config: DeviceConnectionConfig): Promise<Result<void>> {
    try {
      // 实际实现中，这里建立TCP连接
      this.connected = true
      this.commandTimeout = config.timeout || 5000
      return { success: true }
    } catch (error) {
      return { success: false, error: String(error) }
    }
  }

  /**
   * 断开连接
   */
  async disconnect(): Promise<Result<void>> {
    try {
      this.connected = false
      // 清除所有待处理命令
      this.pendingCommands.forEach((pending) => {
        clearTimeout(pending.timer)
        pending.reject(new Error('连接已断开'))
      })
      this.pendingCommands.clear()
      return { success: true }
    } catch (error) {
      return { success: false, error: String(error) }
    }
  }

  /**
   * 检查是否已连接
   */
  isConnected(): boolean {
    return this.connected
  }

  /**
   * 电机使能
   */
  async enable(axis?: B140Axis | B140AxisMask): Promise<void> {
    const cmd = B140CommandBuilder.enable(axis)
    await this.sendCommand(cmd)
  }

  /**
   * 电机关闭
   */
  async disable(axis?: B140Axis | B140AxisMask): Promise<void> {
    const cmd = B140CommandBuilder.disable(axis)
    await this.sendCommand(cmd)
  }

  /**
   * 设置速度
   */
  async setSpeed(axis: B140Axis, speed: number): Promise<void> {
    const cmd = B140CommandBuilder.speed(axis, speed)
    await this.sendCommand(cmd)
  }

  /**
   * 设置绝对位置
   */
  async setPositionAbsolute(axis: B140Axis, position: number): Promise<void> {
    const cmd = B140CommandBuilder.positionAbsolute(axis, position)
    await this.sendCommand(cmd)
  }

  /**
   * 设置相对位置
   */
  async setPositionRelative(axis: B140Axis, distance: number): Promise<void> {
    const cmd = B140CommandBuilder.positionRelative(axis, distance)
    await this.sendCommand(cmd)
  }

  /**
   * 开始运动
   */
  async beginMotion(axis?: B140Axis): Promise<void> {
    const cmd = B140CommandBuilder.beginMotion(axis)
    await this.sendCommand(cmd)
  }

  /**
   * 停止运动
   */
  async stop(axis?: B140Axis): Promise<void> {
    const cmd = B140CommandBuilder.stopMotion(axis)
    await this.sendCommand(cmd)
  }

  /**
   * 紧急停止
   */
  async emergencyStop(): Promise<void> {
    const cmd = B140CommandBuilder.emergencyStop()
    await this.sendCommand(cmd)
  }

  /**
   * 回零
   */
  async home(axis: B140Axis): Promise<void> {
    const cmd = B140CommandBuilder.home(axis)
    await this.sendCommand(cmd)
  }

  /**
   * 查询是否运动中
   */
  async isMoving(): Promise<Result<boolean>> {
    try {
      const response = await this.sendCommand(B140CommandBuilder.tellStatus())
      const status = B140ResponseParser.parseStatus(response)
      if (status) {
        return { success: true, data: status.moving }
      }
      return { success: false, error: '解析状态失败' }
    } catch (error) {
      return { success: false, error: String(error) }
    }
  }

  /**
   * 获取所有轴状态
   */
  async getAllAxesStatus(): Promise<B140AxisStatus[]> {
    const response = await this.sendCommand(B140CommandBuilder.tellPosition())
    const positions = B140ResponseParser.parseMultiAxisPosition(response)
    
    const axes: B140Axis[] = ['A', 'B', 'C', 'D']
    return axes.map(axis => ({
      axis,
      motorPosition: positions.get(axis) || 0,
      isMoving: false,  // 需要通过TS命令查询
      isHomed: false    // 需要通过其他方式判断
    }))
  }

  /**
   * 等待运动完成
   */
  async waitForMotionComplete(axis?: B140Axis, timeout: number = 60000): Promise<void> {
    const start = Date.now()
    while (Date.now() - start < timeout) {
      const moving = await this.isMoving()
      if (moving.success && !moving.data) {
        return
      }
      await this.delay(100)
    }
    throw new Error(`等待运动完成超时: ${axis || '所有轴'}`)
  }

  /**
   * 设置轴方向（是否取反）
   */
  setAxisDirection(axis: B140Axis, reverse: boolean): void {
    this.axisDirections.set(axis, reverse ? -1 : 1)
  }

  /**
   * 延时辅助方法
   */
  private delay(ms: number): Promise<void> {
    return new Promise(resolve => setTimeout(resolve, ms))
  }
}
```

## 九、注意事项

1. **命令结束符**：所有命令必须以 `\r` 结尾（驱动层会自动添加）

2. **轴方向**：可通过 `setAxisDirection()` 设置轴方向取反，位置值会自动乘以方向系数

3. **响应超时**：默认命令超时为5秒，运动等待超时为60秒

4. **状态轮询**：`waitForMotionComplete` 通过100ms间隔轮询 `TS` 命令判断运动状态

5. **错误处理**：所有异步方法都应该使用try-catch进行错误处理

6. **连接管理**：确保在程序退出前调用 `disconnect()` 断开连接

7. **急停使用**：`emergencyStop()` 会立即停止所有轴，不遵循减速曲线

8. **限位保护**：建议配置软限位以防止机械损坏
